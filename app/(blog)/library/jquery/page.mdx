```html
<!-- @jquery -->
<button>click me</button>
<p></p>

<script>
  $('button').on('click', () => {
    let defer = $.Deferred();

    defer
      .then((value) => value * 2)
      .done((value) => {
        $('p').html(value);
      });

    defer.resolve(5);
  });
</script>
```
